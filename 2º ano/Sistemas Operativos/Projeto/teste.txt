Comandos possíveis:

./dclient -a "Romeo and Juliet" "William Shakespeare" "1997" "112.txt"      # indexar
./dclient -c 1                                                              # consultar
./dclient -l 1 "Romeo"                                                      # procurar linhas com palavra
./dclient -s "praia"                                                        # procurar docs com palavra
./dclient -s "praia" "nr_processes"                                         # procurar docs com restrição
./dclient -d 1                                                              # deletar
./dclient -f                                                                # sair do programa
